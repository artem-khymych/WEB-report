                        return {
                            id: Math.random().toString(36).substr(2, 9), // Генеруємо випадковий id
                            name: medicine.name,
                            price: medicine.price * 0.7, // Знижка 30%
                        };
                    } else {
                        return {
                            id: Math.random().toString(36).substr(2, 9), // Генеруємо випадковий id
                            name: medicine.name,
                            price: medicine.price,
                        };
                    }
                });

                return discountedMedicines;
            };
