            <style>
            /* Зміна шрифту для всього тексту */
            body {
                font-family: Arial, sans-serif;
            }

            /* Зміна розміру шрифту та кольору для заголовків */
            h1 {
                font-size: 24px;
                color: #333;
            }

            /* Вирівнювання тексту по центру */
            .centered-text {
                text-align: center;
            }

            /* Зміна кольору тексту */
            .blue-text {
                color: blue;
            }

            /* Створення таблиці з допомогою flexbox */
            .flex-table {
                display: flex;
                flex-direction: column;
                border: 1px solid #ccc;
            }

            /* Стилі для рядків таблиці */
            .flex-row {
                display: flex;
            }

            /* Стилі для комірок таблиці */
            .flex-cell {
                flex: 1;
                padding: 5px;
                border-right: 1px solid #ccc;
            }

            /* Остання комірка не має правої границі */
            .flex-cell:last-child {
                border-right: none;
            }

            /* Зміна фонового кольору сторінки */
            body {
                background-color: #f0f0f0;
            }

            /* Додання контуру елементам */
            .outlined-box {
                border: 1px solid #ccc;
            }

            /* Видалення маркерів списків */
            .no-bullets {
                list-style-type: none;
            }

            /* Псевдокласи */
            a:hover {
                color: red;
            }

            /* Градієнтний фон */
            .gradient-bg {
                background: linear-gradient(to right, #ff7e5f, #feb47b);
            }

            /* Анімація */
            @keyframes example {
                0%   {opacity: 0;}
                100% {opacity: 1;}
            }

            .animated-element {
                animation: example 2s infinite alternate;
            }
        </style>
